//要求：100以内所有整数中9的个数

#define  _CRT_SECURE_NO_WARNINGS 1
#include<stdio.h>
int main()
{
	int i = 0;
	int count = 0;
	for(i = 1;i <=100;i++)
	{
		if (i%10==9)
		{
			printf("%d ", i);
			count++;
		}

		if (i /10 == 9)
		{
			printf("%d ", i);
			count++;
		}
	}
	printf("\ncount=%d", count);
	return 0;
}
