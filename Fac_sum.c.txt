// 求1!+2!+3!的和

#define  _CRT_SECURE_NO_WARNINGS 1
#include<stdio.h>
int main()
{
    int n = 0;
    int i = 0;
    int ret = 1;
    int sum = 0;
    for (n= 1;n<= 3;n++)
    {
        int ret = 1;
        for (i = 1;i <= n;i++)
        {
            ret *= i;
        }
        sum += ret;
    }
    printf("%d", sum);
    return 0;
}
